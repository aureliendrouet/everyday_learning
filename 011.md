# Shader Trick, Gameboy Palette

```c
// Running on shadertoy at
// https://www.shadertoy.com/view/4tsyRB
vec3 palette[4] = vec3[](
vec3(31./255.,31./255.,31./255.),
vec3(77./255.,83./255.,60./255.),
vec3(139./255., 149./255., 109./255.),
vec3(196./255., 207./255., 161./255.)
);

void mainImage( out vec4 fragColor, in vec2 fragCoord )
{
    vec2 uv = (fragCoord.xy / iResolution.xy);
    vec3 gammaColor = texture(iChannel0, uv).xyz;
    // ungamma (should be 2.2 but...)
    vec3 linearColor = pow(gammaColor, vec3(2.0));
    // grayscale conversion
    float gray = dot(linearColor, vec3(0.2126, 0.7152, 0.0722));
    // regamma
    float gammaGray = sqrt(gray);
    // pick from palette
    vec3 color = palette[int(gammaGray * 3. + 0.25)];
    fragColor = vec4(color, 1.0);
}
```
